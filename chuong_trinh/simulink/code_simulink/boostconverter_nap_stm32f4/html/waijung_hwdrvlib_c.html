509int SysTimer_IsTimeout(SYS_TIMER_STRUCT* timer)



510{



511  uint32_t Capture;



512



513  /* Capture current tick */



514  Capture = (uint32_t)systick_count;



515



516  /* Check */



517  if (Capture >= timer->start) {



518    if ((Capture - timer->start) >= timer->length)



519      return 1;



520  } else {

522      return 1;



523  }

525  return 0;

531void SysTimer_uS_Start(SYS_TIMER_uS_STRUCT *timer, uint32_t us)



532{

534  timer->tick = SYS_CURRENT_TICK;



535



536  /* Tick overflow count */

543int SysTimer_uS_IsTimeout(SYS_TIMER_uS_STRUCT* timer)



544{



545  uint32_t curr_tick;



546  uint32_t curr_tid;



547



548  /* Already timeout ? */



549  if (timer->target_us == 0) {



550    return 1;



551  }



552



553  /* Capture current tick */



554  curr_tick = SYS_CURRENT_TICK;



555  curr_tid = systick_count;



556



557  /* Processing TID*/



558  while (timer->tid != curr_tid) {



559    if (timer->target_us > timer->tick) {



560      timer->target_us -= timer->tick;



561      timer->tid ++;



562      timer->tick = SYSTICKRELOADVALUE-1;



563    } else {



564      timer->target_us = 0;



565      return 1;                        /* Timeout */



566    }



567  }



568



569  /* Processing Tick */



570  if ((timer->tick > curr_tick) && ((timer->tick - curr_tick) > timer->target_us))



571  {



572    timer->target_us = 0;



573    return 1;                          /* Timeout */



574  }



575



576  /* Need wait */



577  return 0;



578}

675{

678  DMA_Init(RegularADC_DMA_STREAM, &RegularADC_DMA_InitStructure);



679  DMA_Cmd(RegularADC_DMA_STREAM, ENABLE);



680



681  /* ADC Common */



682  ADC_CommonInit(&RegularADC_ADC_CommonInitStructure);



683



684  /* ADC */



685  ADC_Init(ADC1, &RegularADC_ADC_InitStructure);



686



687  /* ADCx regular channel1 configuration *************************************/



688  ADC_RegularChannelConfig(ADC1, ADC_Channel_0, 1, ADC_SampleTime_3Cycles);



689  ADC_RegularChannelConfig(ADC1, ADC_Channel_1, 2, ADC_SampleTime_3Cycles);



690  ADC_RegularChannelConfig(ADC1, ADC_Channel_4, 3, ADC_SampleTime_3Cycles);



691



692  /* Enable DMA request after last transfer (Single-ADC mode) */



693  ADC_DMARequestAfterLastTransferCmd(ADC1, ENABLE);



694



695  /* Enable ADC1 DMA */



696  ADC_DMACmd(ADC1, ENABLE);



697



698  /* Enable ADC1 */



699  ADC_Cmd(ADC1, ENABLE);



700



701  /* Start ADC Software Conversion */



702  ADC_SoftwareStartConv(ADC1);



703



704  /* Wait for ADA1-DMA transfer complete */



705  while (!DMA_GetFlagStatus(RegularADC_DMA_STREAM, RegularADC_DMA_TC_FLAG)) ;



706



707  /* Clear */



708  DMA_ClearFlag(RegularADC_DMA_STREAM, RegularADC_DMA_TC_FLAG);



709}
