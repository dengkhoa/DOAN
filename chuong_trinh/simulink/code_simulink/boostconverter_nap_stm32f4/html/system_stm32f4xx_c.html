297/**



298 * @brief  Setup the microcontroller system



299 *         Initialize the Embedded Flash Interface, the PLL and update the



300 *         SystemFrequency variable.



301 * @param  None



302 * @retval None



303 */



304void SystemInit(void)



305{



306  /* FPU settings ------------------------------------------------------------*/



307#if (__FPU_PRESENT == 1) && (__FPU_USED == 1)



308



309  SCB->CPACR |= ((3UL << 10*2)|(3UL << 11*2));/* set CP10 and CP11 Full Access */



310



311#endif



312



313  /* Reset the RCC clock configuration to the default reset state ------------*/



314  /* Set HSION bit */



315  RCC->CR |= (uint32_t)0x00000001;



316



317  /* Reset CFGR register */



318  RCC->CFGR = 0x00000000;



319



320  /* Reset HSEON, CSSON and PLLON bits */



321  RCC->CR &= (uint32_t)0xFEF6FFFF;



322



323  /* Reset PLLCFGR register */



324  RCC->PLLCFGR = 0x24003010;



325



326  /* Reset HSEBYP bit */



327  RCC->CR &= (uint32_t)0xFFFBFFFF;



328



329  /* Disable all interrupts */



330  RCC->CIR = 0x00000000;



331



332#if defined(DATA_IN_ExtSRAM) || defined(DATA_IN_ExtSDRAM)



333



334  SystemInit_ExtMemCtl();



335



336#endif                                 /* DATA_IN_ExtSRAM || DATA_IN_ExtSDRAM */



337



338  /* Configure the System clock source, PLL Multiplier and Divider factors,



339     AHB/APBx prescalers and Flash settings ----------------------------------*/



340  SetSysClock();



341



342  /* Configure the Vector Table location add offset address ------------------*/



343#ifdef VECT_TAB_SRAM



344



345  SCB->VTOR = SRAM_BASE | VECT_TAB_OFFSET;/* Vector Table Relocation in Internal SRAM */



346



347#else



348



349  SCB->VTOR = FLASH_BASE | VECT_TAB_OFFSET;/* Vector Table Relocation in Internal FLASH */



350



351#endif



352



353}

355/**



356 * @brief  Update SystemCoreClock variable according to Clock Register Values.



357 *         The SystemCoreClock variable contains the core clock (HCLK), it can



358 *         be used by the user application to setup the SysTick timer or configure



359 *         other parameters.



360 *



361 * @note   Each time the core clock (HCLK) changes, this function must be called



362 *         to update SystemCoreClock variable value. Otherwise, any configuration



363 *         based on this variable will be incorrect.



364 *



365 * @note   - The system frequency computed by this function is not the real



366 *           frequency in the chip. It is calculated based on the predefined



367 *           constant and the selected clock source:



368 *



369 *           - If SYSCLK source is HSI, SystemCoreClock will contain the HSI_VALUE(*)



370 *



371 *           - If SYSCLK source is HSE, SystemCoreClock will contain the HSE_VALUE(**)



372 *



373 *           - If SYSCLK source is PLL, SystemCoreClock will contain the HSE_VALUE(**)



374 *             or HSI_VALUE(*) multiplied/divided by the PLL factors.



375 *



376 *         (*) HSI_VALUE is a constant defined in stm32f4xx.h file (default value



377 *             16 MHz) but the real value may vary depending on the variations



378 *             in voltage and temperature.



379 *



380 *         (**) HSE_VALUE is a constant defined in stm32f4xx.h file (default value



381 *              25 MHz), user has to ensure that HSE_VALUE is same as the real



382 *              frequency of the crystal used. Otherwise, this function may



383 *              have wrong result.



384 *



385 *         - The result of this function could be not correct when using fractional



386 *           value for HSE crystal.



387 *



388 * @param  None



389 * @retval None



390 */



391void SystemCoreClockUpdate(void)



392{



393  uint32_t tmp = 0, pllvco = 0, pllp = 2, pllsource = 0, pllm = 2;



394



395#if defined(STM32F412xG) || defined(STM32F446xx)



396



397  uint32_t pllr = 2;



398



399#endif                                 /* STM32F412xG || STM32F446xx */



400



401  /* Get SYSCLK source -------------------------------------------------------*/



402  tmp = RCC->CFGR & RCC_CFGR_SWS;



403  switch (tmp)



404  {



405   case 0x00:                          /* HSI used as system clock source */



406    SystemCoreClock = HSI_VALUE;



407    break;



408



409   case 0x04:                          /* HSE used as system clock source */



410    SystemCoreClock = HSE_VALUE;



411    break;



412



413   case 0x08:                          /* PLL P used as system clock source */



414    /* PLL_VCO = (HSE_VALUE or HSI_VALUE / PLL_M) * PLL_N



415       SYSCLK = PLL_VCO / PLL_P



416     */



417    pllsource = (RCC->PLLCFGR & RCC_PLLCFGR_PLLSRC) >> 22;



418    pllm = RCC->PLLCFGR & RCC_PLLCFGR_PLLM;



419



420#if defined(STM32F40_41xxx) || defined(STM32F427_437xx) || defined(STM32F429_439xx) || defined(STM32F401xx) || defined(STM32F412xG) || defined(STM32F446xx) || defined(STM32F469_479xx)



421



422    if (pllsource != 0) {



423      /* HSE used as PLL clock source */



424      pllvco = (HSE_VALUE / pllm) * ((RCC->PLLCFGR & RCC_PLLCFGR_PLLN) >> 6);



425    } else {



426      /* HSI used as PLL clock source */



427      pllvco = (HSI_VALUE / pllm) * ((RCC->PLLCFGR & RCC_PLLCFGR_PLLN) >> 6);



428    }



429



430#elif defined(STM32F410xx) || defined(STM32F411xE)



431#if defined(USE_HSE_BYPASS)



432



433    if (pllsource != 0) {



434      /* HSE used as PLL clock source */



435      pllvco = (HSE_BYPASS_INPUT_FREQUENCY / pllm) * ((RCC->PLLCFGR &



436        RCC_PLLCFGR_PLLN) >> 6);



437    }



438



439#else



440



441    if (pllsource == 0) {



442      /* HSI used as PLL clock source */



443      pllvco = (HSI_VALUE / pllm) * ((RCC->PLLCFGR & RCC_PLLCFGR_PLLN) >> 6);



444    }



445



446#endif                                 /* USE_HSE_BYPASS */



447#endif                                 /* STM32F40_41xxx || STM32F427_437xx || STM32F429_439xx || STM32F401xx || STM32F412xG ||  STM32F446xx || STM32F469_479xx */



448



449    pllp = (((RCC->PLLCFGR & RCC_PLLCFGR_PLLP) >>16) + 1 ) *2;



450    SystemCoreClock = pllvco/pllp;



451    break;



452



453#if defined(STM32F412xG) || defined(STM32F446xx)



454



455   case 0x0C:                          /* PLL R used as system clock source */



456    /* PLL_VCO = (HSE_VALUE or HSI_VALUE / PLL_M) * PLL_N



457       SYSCLK = PLL_VCO / PLL_R



458     */



459    pllsource = (RCC->PLLCFGR & RCC_PLLCFGR_PLLSRC) >> 22;



460    pllm = RCC->PLLCFGR & RCC_PLLCFGR_PLLM;



461    if (pllsource != 0) {



462      /* HSE used as PLL clock source */



463      pllvco = (HSE_VALUE / pllm) * ((RCC->PLLCFGR & RCC_PLLCFGR_PLLN) >> 6);



464    } else {



465      /* HSI used as PLL clock source */



466      pllvco = (HSI_VALUE / pllm) * ((RCC->PLLCFGR & RCC_PLLCFGR_PLLN) >> 6);



467    }



468



469    pllr = (((RCC->PLLCFGR & RCC_PLLCFGR_PLLR) >>28) + 1 ) *2;



470    SystemCoreClock = pllvco/pllr;



471    break;



472



473#endif                                 /* STM32F412xG || STM32F446xx */



474



475   default:



476    SystemCoreClock = HSI_VALUE;



477    break;



478  }



479



480  /* Compute HCLK frequency --------------------------------------------------*/



481  /* Get HCLK prescaler */



482  tmp = AHBPrescTable[((RCC->CFGR & RCC_CFGR_HPRE) >> 4)];



483



484  /* HCLK frequency */



485  SystemCoreClock >>= tmp;



486}

488/**



489 * @brief  Configures the System clock source, PLL Multiplier and Divider factors,



490 *         AHB/APBx prescalers and Flash settings



491 * @Note   This function should be called only once the RCC clock configuration



492 *         is reset to the default reset state (done in SystemInit() function).



493 * @param  None



494 * @retval None



495 */



496static void SetSysClock(void)



497{



498



499#if defined(STM32F40_41xxx) || defined(STM32F427_437xx) || defined(STM32F429_439xx) || defined(STM32F401xx) || defined(STM32F412xG) || defined(STM32F446xx)|| defined(STM32F469_479xx)



500



501  /******************************************************************************/



502  /*            PLL (clocked by HSE) used as System clock source                */



503  /******************************************************************************/



504  __IO uint32_t StartUpCounter = 0, HSEStatus = 0;



505



506  /* Enable HSE */



507  RCC->CR |= ((uint32_t)RCC_CR_HSEON);



508



509  /* Wait till HSE is ready and if Time out is reached exit */



510  do {



511    HSEStatus = RCC->CR & RCC_CR_HSERDY;



512    StartUpCounter++;



513  } while ((HSEStatus == 0) && (StartUpCounter != HSE_STARTUP_TIMEOUT));



514



515  if ((RCC->CR & RCC_CR_HSERDY) != RESET) {



516    HSEStatus = (uint32_t)0x01;



517  } else {



518    HSEStatus = (uint32_t)0x00;



519  }



520



521  if (HSEStatus == (uint32_t)0x01) {



522    /* Select regulator voltage output Scale 1 mode */



523    RCC->APB1ENR |= RCC_APB1ENR_PWREN;



524    PWR->CR |= PWR_CR_VOS;



525



526    /* HCLK = SYSCLK / 1*/



527    RCC->CFGR |= RCC_CFGR_HPRE_DIV1;



528



529#if defined(STM32F40_41xxx) || defined(STM32F427_437xx) || defined(STM32F429_439xx) || defined(STM32F412xG) || defined(STM32F446xx) || defined(STM32F469_479xx)



530



531    /* PCLK2 = HCLK / 2*/



532    RCC->CFGR |= RCC_CFGR_PPRE2_DIV2;



533



534    /* PCLK1 = HCLK / 4*/



535    RCC->CFGR |= RCC_CFGR_PPRE1_DIV4;



536



537#endif                                 /* STM32F40_41xxx || STM32F427_437x || STM32F429_439xx  || STM32F412xG || STM32F446xx || STM32F469_479xx */



538



539#if defined(STM32F401xx)



540



541    /* PCLK2 = HCLK / 2*/



542    RCC->CFGR |= RCC_CFGR_PPRE2_DIV1;



543



544    /* PCLK1 = HCLK / 4*/



545    RCC->CFGR |= RCC_CFGR_PPRE1_DIV2;



546



547#endif                                 /* STM32F401xx */



548



549#if defined(STM32F40_41xxx) || defined(STM32F427_437xx) || defined(STM32F429_439xx) || defined(STM32F401xx) || defined(STM32F469_479xx)



550



551    /* Configure the main PLL */



552    RCC->PLLCFGR = PLL_M | (PLL_N << 6) | (((PLL_P >> 1) -1) << 16) |



553      (RCC_PLLCFGR_PLLSRC_HSE) | (PLL_Q << 24);



554



555#endif                                 /* STM32F40_41xxx || STM32F401xx || STM32F427_437x || STM32F429_439xx || STM32F469_479xx */



556



557#if defined(STM32F412xG) || defined(STM32F446xx)



558



559    /* Configure the main PLL */



560    RCC->PLLCFGR = PLL_M | (PLL_N << 6) | (((PLL_P >> 1) -1) << 16) |



561      (RCC_PLLCFGR_PLLSRC_HSE) | (PLL_Q << 24) | (PLL_R << 28);



562



563#endif                                 /* STM32F412xG || STM32F446xx */



564



565    /* Enable the main PLL */



566    RCC->CR |= RCC_CR_PLLON;



567



568    /* Wait till the main PLL is ready */



569    while ((RCC->CR & RCC_CR_PLLRDY) == 0) {



570    }



571



572#if defined(STM32F427_437xx) || defined(STM32F429_439xx) || defined(STM32F446xx) || defined(STM32F469_479xx)



573



574    /* Enable the Over-drive to extend the clock frequency to 180 Mhz */



575    PWR->CR |= PWR_CR_ODEN;



576    while ((PWR->CSR & PWR_CSR_ODRDY) == 0) {



577    }



578



579    PWR->CR |= PWR_CR_ODSWEN;



580    while ((PWR->CSR & PWR_CSR_ODSWRDY) == 0) {



581    }



582



583    /* Configure Flash prefetch, Instruction cache, Data cache and wait state */



584    FLASH->ACR = FLASH_ACR_PRFTEN | FLASH_ACR_ICEN |FLASH_ACR_DCEN |



585      FLASH_ACR_LATENCY_5WS;



586



587#endif                                 /* STM32F427_437x || STM32F429_439xx || STM32F446xx || STM32F469_479xx */



588



589#if defined(STM32F40_41xxx) || defined(STM32F412xG)



590



591    /* Configure Flash prefetch, Instruction cache, Data cache and wait state */



592    FLASH->ACR = FLASH_ACR_PRFTEN | FLASH_ACR_ICEN |FLASH_ACR_DCEN |



593      FLASH_ACR_LATENCY_5WS;



594



595#endif                                 /* STM32F40_41xxx  || STM32F412xG */



596



597#if defined(STM32F401xx)



598



599    /* Configure Flash prefetch, Instruction cache, Data cache and wait state */



600    FLASH->ACR = FLASH_ACR_PRFTEN | FLASH_ACR_ICEN |FLASH_ACR_DCEN |



601      FLASH_ACR_LATENCY_2WS;



602



603#endif                                 /* STM32F401xx */



604



605    /* Select the main PLL as system clock source */



606    RCC->CFGR &= (uint32_t)((uint32_t)~(RCC_CFGR_SW));



607    RCC->CFGR |= RCC_CFGR_SW_PLL;



608



609    /* Wait till the main PLL is used as system clock source */



610    while ((RCC->CFGR & (uint32_t)RCC_CFGR_SWS ) != RCC_CFGR_SWS_PLL) ;



611



612    {



613    }



614  } else {                             /* If HSE fails to start-up, the application will have wrong clock



615                                          configuration. User can add here some code to deal with this error */



616  }



617



618#elif defined(STM32F410xx) || defined(STM32F411xE)



619#if defined(USE_HSE_BYPASS)



620



621  /******************************************************************************/



622  /*            PLL (clocked by HSE) used as System clock source                */



623  /******************************************************************************/



624  __IO uint32_t StartUpCounter = 0, HSEStatus = 0;



625



626  /* Enable HSE and HSE BYPASS */
