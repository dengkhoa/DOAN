657/* Low precision non-Blocking delay timer.



658 ** Relolution of timer is one TICK step interrupt.



659 */



660void SysTimer_Start(SYS_TIMER_STRUCT* timer, uint32_t ms)



661{



662  /* Calculate amount of tick count */



663  timer->length = (uint32_t)((float)ms/(1000*SYS_TICKSTEP));



664



665  /* Capture current tick */



666  timer->start = (uint32_t)systick_count;



667}



668



669int SysTimer_IsTimeout(SYS_TIMER_STRUCT* timer)



670{



671  uint32_t Capture;



672



673  /* Capture current tick */



674  Capture = (uint32_t)systick_count;



675



676  /* Check */



677  if (Capture >= timer->start) {



678    if ((Capture - timer->start) >= timer->length)



679      return 1;



680  } else {



681    if (((0xFFFFFFFF - timer->start)+Capture) >= timer->length)



682      return 1;



683  }



684



685  return 0;



686}



687



688/* Improved precision non-Blocking delay timer.



689 ** Relolution of timer is 1uS.



690 */



691void SysTimer_uS_Start(SYS_TIMER_uS_STRUCT *timer, uint32_t us)



692{



693  /* Tick */



694  timer->tick = SYS_CURRENT_TICK;



695



696  /* Tick overflow count */



697  timer->tid = (uint32_t)systick_count;



698



699  /* Target */



700  timer->target_us = us*((uint32_t)HCLK/1000000);



701}



702



703int SysTimer_uS_IsTimeout(SYS_TIMER_uS_STRUCT* timer)



704{



705  uint32_t curr_tick;



706  uint32_t curr_tid;



707



708  /* Already timeout ? */



709  if (timer->target_us == 0) {



710    return 1;



711  }



712



713  /* Capture current tick */



714  curr_tick = SYS_CURRENT_TICK;



715  curr_tid = systick_count;



716



717  /* Processing TID*/



718  while (timer->tid != curr_tid) {



719    if (timer->target_us > timer->tick) {



720      timer->target_us -= timer->tick;



721      timer->tid ++;



722      timer->tick = SYSTICKRELOADVALUE-1;



723    } else {



724      timer->target_us = 0;



725      return 1;                        /* Timeout */



726    }



727  }



728



729  /* Processing Tick */



730  if ((timer->tick > curr_tick) && ((timer->tick - curr_tick) > timer->target_us))



731  {



732    timer->target_us = 0;



733    return 1;                          /* Timeout */



734  }



735



736  /* Need wait */



737  return 0;



738}



739



740/* Blocking delay, uS */



741void SysTimer_delay_us(uint32_t us)



742{



743  SYS_TIMER_uS_STRUCT timer;



744  SysTimer_uS_Start(&timer, us);



745  while (SysTimer_uS_IsTimeout(&timer) == 0) ;



746}



747



748/* Blocking delay, mS */



749void SysTimer_delay_ms(uint32_t ms)



750{



751  SYS_TIMER_STRUCT timer;



752  SysTimer_Start(&timer, ms);



753  while (SysTimer_IsTimeout(&timer) == 0) ;



754}

829  CRC_ResetDR();



830  profiler_records[profilerCRCLength] = CRC_CalcBlockCRC((uint32_t *)



831    profiler_records, profilerCRCLength);

833  /* Write to DMA buffer */



834  /* Update overflow counter */



835  /* Setup */

841  DMA_Cmd(DMA1_Stream3, ENABLE);



842

855  USART_DMACmd(USART3, USART_DMAReq_Tx, DISABLE);
