164/* Function: rtOneStep ========================================================



165 *



166 * Abstract:



167 *   Perform one step of the model.  This function is modeled such that



168 *   it could be called from an interrupt service routine (ISR) with minor



169 *   modifications.



170 */



171static void rt_OneStep(void)



172{



173    /* Disable interrupts here */



174



175    /***********************************************



176     * Check and see if base step time is too fast *



177     ***********************************************/



178    if (OverrunFlags[0]++) {



179        rtmSetErrorStatus(RT_MDL, "Overrun");



180    }



181



182    /*************************************************



183     * Check and see if an error status has been set *



184     * by an overrun or by the generated code.       *



185     *************************************************/



186    if (rtmGetErrorStatus(RT_MDL) != NULL) {



187        return;



188    }



189



190    /* Save FPU context here (if necessary) */



191    /* Reenable interrupts here */



192    /* Set model inputs here */



193



194    /**************



195     * Step model *



196     **************/



197    MODEL_STEP();



198



199    /* Get model outputs here */



200



201    /**************************



202     * Decrement overrun flag *



203     **************************/



204    OverrunFlags[0]--;



205



206    rtExtModeCheckEndTrigger();



207



208    /* Disable interrupts here */



209    /* Restore FPU context here (if necessary) */



210    /* Reenable interrupts here */



211



212} /* end rtOneStep */



213



214#else /* multitask */

216/* Function: rtOneStep ========================================================



217 *



218 * Abstract:



219 *   Perform one step of the model.  This function is modeled such that



220 *   it could be called from an interrupt service routine (ISR) with minor



221 *   modifications.



222 *



223 *   Note that error checking is only necessary when this routine is



224 *   attached to an interrupt.



225 *



226 *   Also, you may wish to unroll any or all of for and while loops to



227 *   improve the real-time performance of this function.



228 */



229static void rt_OneStep(void)



230{



231    int_T i;



232



233    /* Disable interrupts here */



234



235    /***********************************************



236     * Check and see if base step time is too fast *



237     ***********************************************/



238    if (OverrunFlags[0]++) {



239        rtmSetErrorStatus(RT_MDL, "Overrun");



240    }



241



242    /*************************************************



243     * Check and see if an error status has been set *



244     * by an overrun or by the generated code.       *



245     *************************************************/



246    if (rtmGetErrorStatus(RT_MDL) != NULL) {



247        return;



248    }



249



250    /* Save FPU context here (if necessary) */



251    /* Reenable interrupts here */



252    



253    /*************************************************



254     * Update EventFlags and check subrate overrun   *



255     *************************************************/



256    for (i = FIRST_TID+1; i < NUMST; i++) {



257        if (rtmStepTask(RT_MDL,i) && eventFlags[i]++) {



258            OverrunFlags[0]--;



259            OverrunFlags[i]++;



260            /* Sampling too fast */



261            rtmSetErrorStatus(RT_MDL, "Overrun");



262            return;



263        }



264        if (++rtmTaskCounter(RT_MDL,i) == rtmCounterLimit(RT_MDL,i))



265            rtmTaskCounter(RT_MDL, i) = 0;



266    }



267



268    /* Set model inputs associated with base rate here */



269



270    /*******************************************



271     * Step the model for the base sample time *



272     *******************************************/



273    MODEL_STEP(0);



274



275    /* Get model outputs associated with base rate here */



276



277    /************************************************************************



278     * Model step complete for base sample time, now it is okay to          *



279     * re-interrupt this ISR.                                               *



280     ************************************************************************/



281    OverrunFlags[0]--;



282



283    /*********************************************************



284     * Step the model for any other sample times (subrates) *



285     *********************************************************/



286    for (i = FIRST_TID+1; i < NUMST; i++) {



287        /*************************************************************



288         * If task "i" is running, don't run any lower priority task *



289         *************************************************************/



290        if (OverrunFlags[i]) return; 



291



292        if (eventFlags[i]) {



293            OverrunFlags[i]++;



294



295            /* Set model inputs associated with subrate here */



296



297            /******************************************



298             * Step the model for sample time "i" *



299             ******************************************/



300            MODEL_STEP(i);



301



302            /* Get model outputs associated with subrate here */



303            



304            /**********************************************



305             * Indicate task complete for sample time "i" *



306             **********************************************/



307            OverrunFlags[i]--;



308            eventFlags[i]--;



309        }



310    }



311



312    rtExtModeCheckEndTrigger();



313    /* Disable interrupts here */



314    /* Restore FPU context here (if necessary) */



315    /* Enable interrupts here */



316



317} /* end rtOneStep */



318



319#endif /* MULTITASKING */

321/* Function: rt_InitModel ====================================================



322 * 



323 * Abstract: 



324 *   Initialized the model and the overrun flags



325 *



326 */



327static void rt_InitModel(void)



328{



329#if defined(MULTITASKING)



330    int i;



331    for(i=0; i < NUMST; i++) {



332        OverrunFlags[i] = 0;



333        eventFlags[i] = 0;



334    }



335#else



336    OverrunFlags[0] = 0;



337    eventFlags[0] = 0;



338#endif



339



340    /************************



341     * Initialize the model *



342     ************************/



343    MODEL_INITIALIZE();



344}

346/* Function: rt_TermModel ====================================================



347 * 



348 * Abstract:



349 *   Terminates the model and prints the error status



350 *



351 */



352static int_T rt_TermModel(void)



353{



354    MODEL_TERMINATE();



355    



356    {



357        const char_T *errStatus = (const char_T *) (rtmGetErrorStatus(RT_MDL));



358        int_T i = 0;



359        



360        if (errStatus != NULL && strcmp(errStatus, "Simulation finished")) {



361            (void)printf("%s\n", errStatus);



362#if defined(MULTITASKING)



363            for (i = 0; i < NUMST; i++) {



364                if (OverrunFlags[i]) {



365                    (void)printf("ISR overrun - sampling rate too"



366                                 "fast for sample time index %d.\n", i);



367                }



368            }



369#else



370           if (OverrunFlags[i]) { 



371               (void)printf("ISR overrun - base sampling rate too fast.\n");



372           }



373#endif



374            return(1);



375        }



376    }



377    



378    return(0);



379}

381/* Function: main =============================================================



382 *



383 * Abstract:



384 *   Execute model on a generic target such as a workstation.



385 */



386int_T main(int_T argc, const char *argv[])



387{



388    /* External mode */



389    rtParseArgsForExtMode(argc, argv);



390 



391    /*******************************************



392     * warn if the model will run indefinitely *



393     *******************************************/



394#if MAT_FILE==0 && EXT_MODE==0



395    printf("warning: the simulation will run with no stop time; "



396           "to change this behavior select the 'MAT-file logging' option\n");



397    fflush(NULL);



398#endif



399



400    (void)printf("\n** starting the model **\n");



401



402    /************************



403     * Initialize the model *



404     ************************/



405    rt_InitModel();



406



407    /* External mode */



408    rtSetTFinalForExtMode(&rtmGetTFinal(RT_MDL));



409    rtExtModeCheckInit(NUMST);



410    rtExtModeWaitForStartPkt(rtmGetRTWExtModeInfo(RT_MDL),



411                             NUMST,



412                             (boolean_T *)&rtmGetStopRequested(RT_MDL));



413



414    /***********************************************************************



415     * Execute (step) the model.  You may also attach rtOneStep to an ISR, *



416     * in which case you replace the call to rtOneStep with a call to a    *



417     * background task.  Note that the generated code sets error status    *



418     * to "Simulation finished" when MatFileLogging is specified in TLC.   *



419     ***********************************************************************/



420    while (rtmGetErrorStatus(RT_MDL) == NULL &&



421           !rtmGetStopRequested(RT_MDL)) {



422



423        rtExtModePauseIfNeeded(rtmGetRTWExtModeInfo(RT_MDL),



424                               NUMST,



425                               (boolean_T *)&rtmGetStopRequested(RT_MDL));



426



427        if (rtmGetStopRequested(RT_MDL)) break;



428



429        /* external mode */



430        rtExtModeOneStep(rtmGetRTWExtModeInfo(RT_MDL),



431                         NUMST,



432                         (boolean_T *)&rtmGetStopRequested(RT_MDL));



433        



434        rt_OneStep();



435    }



436



437    /*******************************



438     * Cleanup and exit (optional) *



439     *******************************/



440



441#ifdef UseMMIDataLogging



442    rt_CleanUpForStateLogWithMMI(rtmGetRTWLogInfo(RT_MDL));



443#endif



444    rt_StopDataLogging(MATFILE,rtmGetRTWLogInfo(RT_MDL));



445



446    rtExtModeShutdown(NUMST);



447



448    return rt_TermModel();



449}
