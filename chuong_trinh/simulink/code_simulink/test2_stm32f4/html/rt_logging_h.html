38typedef struct MatrixData_Tag {



39  char_T         name[mxMAXNAM];     /* Name of the variable                  */



40  int_T          nRows;              /* number of rows                        */



41  int_T          nCols;              /* number of columns                     */



42  int_T          nDims;              /* number of dimensions                  */



43  int_T          _dims[2];           /* most cases, dimensions are 2          */



44  int_T          *dims;              /* dimensions of the log variable we 



45                                        write to at each simulation time step.



46                                        E.g: (1) Non-frame data - 



47                                                 Signal dimension = [2 X 3]



48                                                 numDims = 2



49                                                 dims[0] = 2, dims[1] = 3



50                                             (2) Frame data - 



51                                                 Signal dimension = [2 X 3]



52                                                 numDims = 1



53                                                 dims[0] = 3                  */



54  void           *re;                /* pointer to real part of the data      */



55  void           *im;                /* pointer to imaginary part, if complex */



56  DTypeId        dTypeID;            /* data type id                          */



57  size_t         elSize;             /* element size in bytes                 */



58



59  RTWLogDataTypeConvert dataTypeConvertInfo;



60



61  mxClassID      mxID;               /* mxId corresponding to this dTypeID    */



62  uint32_T       logical;            /* is this a logical array ?             */



63  uint32_T       complex;            /* is this a complex matrix?             */



64  uint32_T       frameData;          /* is this data frame based?             */



65  uint32_T       frameSize;          /* is this data frame based?             */



66} MatrixData;

68typedef struct ValDimsData_Tag {



69  char_T         name[mxMAXNAM];     /* Name of the variable                  */



70  int_T          nRows;              /* number of rows                        */



71  int_T          nCols;              /* number of columns                     */



72  void           **currSigDims;      /* dimensions of current output          */



73  int_T          *currSigDimsSize;   /* size in bytes of current sig dims     */



74  real_T         *dimsData;          /* pointer to the value of dimension     */



75} ValDimsData;



76



77struct LogVar_Tag {



78    MatrixData  data;                 /* Container for name, data etc.,       */



79    ValDimsData *valDims;             /* field of valueDimensions



80                                       1. If all logging signals are fixed-size,



81                                          then we set this field to NULL;



82                                       2. If any logging signal is variable-size,



83                                          then this field will be needed:



84                                         1) For fixed-size signal, this field is



85                                            an empty matrix;



86                                         2) Otherwise, it contains the dimension



87                                            information of the logging signal.



88                                      */



89    int_T      rowIdx;                /* current row index                    */



90    int_T      wrapped;               /* number of times the circular buffer



91                                       * has wrapped around                   */



92    int_T     nDataPoints;            /* total number of data points logged   */



93    int_T     usingDefaultBufSize;    /* used to print a message at end       */



94    int_T     okayToRealloc;          /* reallocate during sim?               */



95    int_T     decimation;             /* decimation factor                    */



96    int_T     numHits;                /* decimation hit count                 */



97



98    int_T     *coords;



99    int_T     *strides;



100    int_T     *currStrides;           /* coords, strides and currStrides will be



101                                         needed when logging variable-size 



102                                         signal to calculate whether the 



103                                         currently logging value is in the range.



104                                         If the current signal is fixed-size,



105                                         these pointers will be set to NULLs;



106                                         otherwise, we allocate memory for them.



107                                         (the size will be nDims in this case)



108                                      */



109



110    LogVar    *next;



111};

156/* status of logging "valueDimensions" field */



157/* 



158  NO_LOGVALDIMS: 



159                 No need to log valueDimensions: 



160                 All signals are fixed-sized.



161



162  LOGVALDIMS_EMPTYMX: 



163                 Signals with mixed dimension modes,



164                 and the signal logged currently



165                 is fixed-sized. So set valueDimensions



166                 field to an empty matrix.



167



168  LOGVALDIMS_VARDIMS:



169                 Signal logged currently is variable-sized.



170*/



171typedef enum {



172    NO_LOGVALDIMS,      



173    LOGVALDIMS_EMPTYMX,



174    LOGVALDIMS_VARDIMS



175} LogValDimsStat;
